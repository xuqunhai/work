var glob = require('glob');
// 获取指定路径下的入口文件
function getEntries(globPath) {
    var files = glob.sync(globPath),
        entries = {};
 
    files.forEach(function(filepath) {
        var split = filepath.split('/');
        var name = split[split.length - 2];
        var entity = [];
        entity.push('./' + filepath);
        entries[name] = entity;
    });
 
    return entries;
}
