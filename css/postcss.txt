postcss postcss-loader postcss-preset-env
引入的其实就是一个处理css兼容性问题的加载器，类似js的babel；

{
    test: /\.less$/,
    use: [
        'style-loader',
        'css-loader',
        {
            loader: 'postcss-loader',
            options: {
                postcssOptions: {
                    plugins: [
                        [
                            'postcss-preset-env',
                            {
                                browsers: 'last 2 versions'
                            }
                        ]
                    ]
                }
            }
        },
        'less-loader'
    ]
}
