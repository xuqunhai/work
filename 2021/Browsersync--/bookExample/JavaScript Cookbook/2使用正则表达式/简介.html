正则表达式是用来查找与给定模式匹配的文本的搜索模式 正则表达式用来匹配颜色值： //查找#a0b1c2 if(result = /#([a-fA-F0-9]{2})([a-fA-F0-9]{2})([a-fA-F0-9]{2})/.exec(color)){ return [parseInt(result[1],16),parseInt(result[2],16),parseInt(result[3],16)]; } //查找#fff if(result = /#([a-fA-F0-9])([a-fA-F0-9])([a-fA-F0-9])/.exec(color)){ return [parseInt(result[1]+result[1],16),parseInt(result[2]+result[2],16),parseInt(result[3]+result[3],16)]; } //查找rgba(0,0,0,0) if(result = /rgba\(0,0,0,0\)/.exec(color)){ return colors['transparent']; } //处理一个命名的颜色 return colors[$.trim(color).toLowerCase()]; JS中，正则表达式通过RegExp对象来管理； RegExp直接量 var re = /regular expression/; 当表达式不会修改的时候使用直接量更高效 作为对象的RegExp var re = new RegExp("Shelley\s+Powers"); 正则的特殊字符 字符 匹配 . 除了换行以外的任何字符 [^...] 方括号中字符以外的任何字符 \b 匹配边界上的单词 \d 从0到9的数字 \w 匹配单词字符（字母、数字、下划线） \n 匹配一个换行 \s 匹配一个空白字符
